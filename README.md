function makeGroupMatches(){
  var matches=[];
  Object.keys(GROUPS).forEach(function(g){
    var ts=GROUPS[g];
    for(var i=0;i<ts.length;i++){
      for(var j=i+1;j<ts.length;j++){
        matches.push({id:g+'_'+i+'_'+j,group_name:g,home_name:ts[i].n,home_flag:ts[i].f,away_name:ts[j].n,away_flag:ts[j].f,home_score:null,away_score:null,played:false});
      }
    }
  });
  return matches;
}
function makeKnockout(){
  var rows=[];
  var rounds={r32:16,r16:8,qf:4,sf:2,fin:1};

    for(var i=0;i<rounds[r];i++){
      rows.push({id:r+'_'+i,round:r,idx:i,home:null,away:null,winner:null,runner_up:null});
    }
  });
  return rows;
}

function esc(s){return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}
function takenAnySet(excludeId){
  // Only non-top10 countries are exclusive
  var s=new Set();
  teams.forEach(function(t){
    if(t.id!==excludeId){
      (t.countries||[]).forEach(function(c){
        if(!isTop10(c.n))s.add(c.n);
      });
    }
  });
  return s;
}

// ── LOAD ALL DATA ─────────────────────────────────────────────────────────────
async function loadAll(){
  try{
    // Load teams
    var dbTeams=await sbGet('teams','order=id');
    if(dbTeams.length===0){
      // First time — seed 9 teams
      var seedTeams=[];
      for(var i=0;i<9;i++) seedTeams.push({id:i,name:'Team '+(i+1),mascot:MASCOTS[i],countries:[],player:'',points:0});
      await sbUpsert('teams',seedTeams);
      teams=seedTeams;
    } else {
      teams=dbTeams.map(function(t){return{id:t.id,name:t.name,mascot:t.mascot,countries:t.countries||[],player:t.player||'',points:t.points||0};});
    }

    // Load group matches
    var dbMatches=await sbGet('group_matches','order=id');
    if(dbMatches.length===0){
      var seed=makeGroupMatches();
      await sbUpsert('group_matches',seed);
      groupMatches=seed;
    } else {
      groupMatches=dbMatches;
    }

    // Load knockout
    var dbKo=await sbGet('knockout','order=round,idx');
    if(dbKo.length===0){
      var seedKo=makeKnockout();
      await sbUpsert('knockout',seedKo);
      knockout=seedKo;
    } else {
      knockout=dbKo;
    }

    // Load smack talk
    smackList=await sbGet('smack_talk','order=created_at');

  } catch(e){
    toast('Could not connect to database. Check your internet connection.', true);
    console.error(e);
  }
}

// ── SCORING ENGINE ────────────────────────────────────────────────────────────
function calcPoints(){
  var result={};
  teams.forEach(function(t){result[t.id]={total:0,breakdown:[]};});
  function addPts(countryName,pts,reason){
    teams.forEach(function(t){
      (t.countries||[]).forEach(function(c){
        if(c.n===countryName){result[t.id].total+=pts;result[t.id].breakdown.push(flagFor(countryName)+' '+countryName+': +'+pts+' ('+reason+')');}
      });
    });
  }
  groupMatches.forEach(function(m){
    if(!m.played||m.home_score===null||m.away_score===null)return;
    var hs=+m.home_score,as=+m.away_score;
    if(hs>as){addPts(m.home_name,PTS.groupWin,'group win');addPts(m.away_name,0,'');}
    else if(as>hs){addPts(m.away_name,PTS.groupWin,'group win');addPts(m.home_name,0,'');}
    else{addPts(m.home_name,PTS.groupDraw,'group draw');addPts(m.away_name,PTS.groupDraw,'group draw');}
  });
  var rdPts={r32:PTS.r32,r16:PTS.r16,qf:PTS.qf,sf:PTS.sf};
  knockout.forEach(function(m){
    if(!m.winner)return;
    if(rdPts[m.round])addPts(m.winner,rdPts[m.round],m.round+' win');
    if(m.round==='fin'){
      addPts(m.winner,PTS.champion,'WORLD CHAMPION! 🏆');
      if(m.runner_up)addPts(m.runner_up,PTS.runnerUp,'Runner-up');
    }
  });
  return result;
}

  // Group match save
  el.querySelectorAll('.save-match-btn').forEach(function(b){
    b.addEventListener('click',async function(){
      var mid=this.dataset.mid;
      var hIn=el.querySelector('.score-input[data-mid="'+mid+'"][data-side="home"]');
      var aIn=el.querySelector('.score-input[data-mid="'+mid+'"][data-side="away"]');
      if(!hIn||!aIn)return;
      var hs=hIn.value.trim(),as=aIn.value.trim();
      if(hs===''||as===''){toast('Please enter both scores before saving.',true);return;}
      try{
        await sbUpdate('group_matches','id=eq.'+mid,{home_score:+hs,away_score:+as,played:true});
        var m=groupMatches.find(function(x){return x.id===mid;});
        if(m){m.home_score=+hs;m.away_score=+as;m.played=true;}
        toast('✅ Score saved!');
        renderResults();
      }catch(e){toast('Error saving score.',true);console.error(e);}
    });
  });

  el.querySelectorAll('select[data-ko]').forEach(function(sel){
    sel.addEventListener('change',async function(){
      var rnd=this.dataset.ko,idx=+this.dataset.idx,side=this.dataset.side,val=this.value||null;
      var m=knockout.find(function(x){return x.round===rnd&&x.idx===idx;});
      if(!m)return;
      m[side]=val;
      if(side!=='winner'&&m.winner&&m.winner!==m.home&&m.winner!==m.away)m.winner=null;
      var upd={};upd[side]=val;
      try{
        await sbUpdate('knockout','id=eq.'+m.id,upd);
        toast('✅ Saved!');
        renderResults();
      }catch(e){toast('Error saving.',true);console.error(e);}
    });
  });

  // Final selects
  var fc=document.getElementById('fin-champion');
  var fr=document.getElementById('fin-runnerup');
  async function saveFinal(){
    var fin=knockout.find(function(m){return m.round==='fin';});
    if(!fin)return;
    try{
      await sbUpdate('knockout','id=eq.'+fin.id,{winner:fin.winner||null,runner_up:fin.runner_up||null});
      toast('✅ Final saved!');
      renderResults();
    }catch(e){toast('Error saving final.',true);console.error(e);}
  }
  if(fc)fc.addEventListener('change',function(){var fin=knockout.find(function(m){return m.round==='fin';});if(fin){fin.winner=this.value||null;saveFinal();}});
  if(fr)fr.addEventListener('change',function(){var fin=knockout.find(function(m){return m.round==='fin';});if(fin){fin.runner_up=this.value||null;saveFinal();}});
}

// ── STANDINGS ─────────────────────────────────────────────────────────────────
function renderStandings(){
  var el=document.getElementById('pg-standings');
  showLoading('standings');
  // Reload from DB to get latest
  Promise.all([
    sbGet('teams','order=id'),
    sbGet('group_matches','order=id'),
    sbGet('knockout','order=round,idx')
  ]).then(function(results){
    teams=results[0].map(function(t){return{id:t.id,name:t.name,mascot:t.mascot,countries:t.countries||[],player:t.player||'',points:t.points||0};});
    groupMatches=results[1];
    knockout=results[2];
    var scored=calcPoints();
    var sorted=teams.slice().sort(function(a,b){return scored[b.id].total-scored[a.id].total;});
    var allPicked=teams.every(function(t){return t.countries.length===2;});
    var html='<div class="stitle">🏆 Standings</div><div class="ssub">Live leaderboard · Points calculated from match results</div>';
    if(!allPicked)html+='<div style="background:rgba(245,197,24,.1);border:1px solid var(--gold);border-radius:10px;padding:13px 18px;margin-bottom:18px;color:var(--gold);font-size:.88rem;">⚠️ Not all teams have picked their 2 countries yet!</div>';
    html+='<div style="display:flex;gap:10px;margin-bottom:18px;flex-wrap:wrap"><button class="btn btn-gold" id="celBtn">🎉 Celebrate!</button><button class="btn btn-outline" id="refreshBtn">🔄 Refresh</button></div>';
    html+='<div class="card"><table><thead><tr><th style="width:46px">#</th><th>Team</th><th style="text-align:right">Points</th></tr></thead><tbody>';
    sorted.forEach(function(t,i){
      var sc=scored[t.id];
      var rc=i===0?'r1':i===1?'r2':i===2?'r3':'';
      html+='<tr><td><div class="rank '+rc+'">'+(i+1)+'</div></td><td><div style="display:flex;align-items:center;gap:10px"><span style="font-size:1.6rem">'+t.mascot+'</span><div><div class="tname">'+esc(t.name)+'</div><div class="cpills">';
      t.countries.forEach(function(c){html+='<span class="pill'+(isTop10(c.n)?' top10':'')+'">'+c.f+' '+esc(c.n)+(isTop10(c.n)?' 🌟':'')+'</span>';});
      if(!t.countries.length)html+='<span class="pill" style="color:var(--red)">No countries yet</span>';
      html+='</div>'+(t.player?'<div><span class="player-pill">⭐ '+esc(t.player)+'</span></div>':'')+'</div></div></td>';
      html+='<td><div class="pts">'+sc.total+' pts</div>';
      if(sc.breakdown.length)html+='<div class="pts-breakdown">'+sc.breakdown.map(function(b){return esc(b);}).join('<br>')+'</div>';
      html+='</td></tr>';
    });
    html+='</tbody></table></div>';
    el.innerHTML=html;
    document.getElementById('celBtn').addEventListener('click',function(){confetti(120);});
    document.getElementById('refreshBtn').addEventListener('click',function(){renderStandings();});
  }).catch(function(e){
    el.innerHTML='<div class="empty"><div class="ei">⚠️</div><h3>Could not load standings</h3><p>Check your internet connection and try again.</p></div>';
    console.error(e);
  });
}

function renderFacts(){
  var el=document.getElementById('pg-facts');
  el.innerHTML='<div class="stitle">🌎 Country Facts & Recipes</div><div class="ssub">All 48 nations · Football history & traditional recipes</div><input class="fsearch" id="fsearch" placeholder="🔍 Search countries..." value="'+esc(factsQ)+'"><div class="facts-grid" id="facts-grid"></div>';

  renderFactsGrid();
}
function renderFactsGrid(){
  var grid=document.getElementById('facts-grid');if(!grid)return;
  var q=factsQ.toLowerCase();var html='';
  ALL.forEach(function(c){
    if(q&&!c.n.toLowerCase().includes(q))return;
    var d=FACTS[c.n];if(!d)return;
    var tab=factTabs[c.n]||'fact';var top=isTop10(c.n);
    html+='<div class="fcard"><div class="fhead"><span class="fflag">'+c.f+'</span><span class="fcname">'+esc(c.n)+'</span>'+(top?'<span class="ftop10-tag">🌟 TOP 10</span>':'')+'</div>';
    html+='<div class="ftabs"><button class="ftab'+(tab==='fact'?' active':'')+'" data-cn="'+esc(c.n)+'" data-tab="fact">⚽ Football</button><button class="ftab'+(tab==='recipe'?' active':'')+'" data-cn="'+esc(c.n)+'" data-tab="recipe">🍽️ Recipe</button></div>';
    html+='<div class="fbody">';
    if(tab==='fact')html+='<p>'+esc(d.fact)+'</p>';
    else html+='<div class="rtitle">'+d.recipe.name+'</div><ul class="ilist">'+d.recipe.ing.map(function(i){return'<li>'+esc(i)+'</li>';}).join('')+'</ul><p class="rsteps">'+esc(d.recipe.steps)+'</p>';
    html+='</div></div>';
  });
  if(!html)html='<div class="empty"><div class="ei">🔍</div><h3>No countries found</h3><p>Try a different search</p></div>';
  grid.innerHTML=html;
  grid.querySelectorAll('.ftab').forEach(function(b){b.addEventListener('click',function(){factTabs[this.dataset.cn]=this.dataset.tab;renderFactsGrid();});});
}

// ── SMACK TALK ────────────────────────────────────────────────────────────────
function renderSmack(){
  var el=document.getElementById('pg-smack');
  showLoading('smack');
  sbGet('smack_talk','order=created_at').then(function(rows){
    smackList=rows;
    var html='<div class="stitle">💬 Smack Talk Board</div><div class="ssub">Family trash talk · Saved for everyone to see!</div><div class="sfeed">';
    if(!smackList.length)html+='<div class="empty"><div class="ei">💬</div><h3>Silence...</h3><p>Be the first to talk smack</p></div>';
    else smackList.slice().reverse().forEach(function(msg){
      var t=teams.find(function(x){return x.id===msg.team_id;})||teams[0];
      var when=new Date(msg.created_at).toLocaleString();
      html+='<div class="sbubble"><div class="shead"><span class="smascot">'+(t?t.mascot:'⚽')+'</span><span class="steam">'+esc(t?t.name:'Team')+'</span><span class="stime">'+esc(when)+'</span><button class="sdel" data-id="'+msg.id+'">🗑️</button></div><div class="stext">'+esc(msg.text)+'</div></div>';
    });
    html+='</div>';
    html+='<div class="card" style="margin-bottom:14px"><div style="font-family:Barlow Condensed,sans-serif;font-size:.78rem;letter-spacing:2px;color:var(--gray);text-transform:uppercase;margin-bottom:11px">Drop your message</div>';
    html+='<div class="scompose"><select class="ssel" id="sselect">';
    teams.forEach(function(t){html+='<option value="'+t.id+'">'+t.mascot+' '+esc(t.name)+'</option>';});
    html+='</select><input class="sinput" id="sinput" placeholder="Talk your smack..." maxlength="140"><button class="btn btn-green" id="ssend">Send 🔥</button></div></div>';
    el.innerHTML=html;

    async function send(){
      var txt=document.getElementById('sinput').value.trim();
      var tid=+document.getElementById('sselect').value;
      if(!txt)return;
      try{
        await sbInsert('smack_talk',{team_id:tid,text:txt});
        toast('💬 Message sent!');
        renderSmack();
      }catch(e){toast('Error sending message.',true);console.error(e);}
    }
    document.getElementById('ssend').addEventListener('click',send);
    document.getElementById('sinput').addEventListener('keydown',function(e){if(e.key==='Enter')send();});
    el.querySelectorAll('.sdel').forEach(function(b){
      b.addEventListener('click',async function(){
        var id=this.dataset.id;
        try{
          await sbDelete('smack_talk','id=eq.'+id);
          renderSmack();
        }catch(e){toast('Error deleting message.',true);}
      });
    });
  }).catch(function(e){
    el.innerHTML='<div class="empty"><div class="ei">⚠️</div><h3>Could not load messages</h3><p>Check your connection.</p></div>';
  });
}
